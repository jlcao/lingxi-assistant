**版本：** v1.0

**参考：** OpenClaw Memory 系统

## 1. 概述

长期记忆功能让灵犀助手能够记住用户偏好、重要事实、历史决策等信息，并在需要时通过语义搜索快速检索。

### 1.1 核心目标

- ✅ 记住用户偏好和习惯
- ✅ 存储重要事实和决策
- ✅ 支持语义搜索和检索
- ✅ 自动提取和更新记忆
- ✅ 工作目录隔离

### 1.2 设计原则

- **简单优先**：先实现文本记忆，后支持向量检索
- **复用现有**：复用 SessionManager 和事件机制
- **渐进增强**：从 MEMORY.md 文件开始，逐步支持向量数据库
- **隐私保护**：记忆数据本地存储，不上传云端

#!/usr/bin/env python3
"""记忆提取器 - 从对话中自动提取重要信息"""

import re
import logging
from typing import List, Dict, Any
from .memory_manager import Memory, MemoryManager

logger = logging.getLogger(__name__)


class MemoryExtractor:
    """记忆提取器"""
    
    def __init__(self, memory_manager: MemoryManager):
        self.memory_manager = memory_manager
        self.extraction_patterns = self._load_patterns()
    
    def _load_patterns(self) -> Dict[str, str]:
        """加载提取模式"""
        return {
            # 用户偏好
            "preference": r"(我喜欢|我讨厌|我习惯|我经常|我很少|我不喜欢).+",
            # 重要事实
            "fact": r"(记住|重要的是|关键是|注意).+",
            # 决策
            "decision": r"(决定了|确定了|就这样吧|采用).+",
            # 待办事项
            "todo": r"(我要|我需要|记得|别忘了).+"
        }
    
    def extract_from_message(self, message: str, role: str = "user") -> List[Memory]:
        """从单条消息中提取记忆"""
        memories = []
        
        for category, pattern in self.extraction_patterns.items():
            matches = re.findall(pattern, message, re.IGNORECASE)
            for match in matches:
                memory = Memory(
                    id=Memory.compute_id(match, category),
                    content=match.strip(),
                    category=category,
                    tags=[],
                    importance=3,
                    created_at=datetime.now().timestamp(),
                    updated_at=datetime.now().timestamp(),
                    access_count=0,
                    workspace_path=self.memory_manager.workspace_path,
                    metadata={"source": "conversation", "role": role}
                )
                memories.append(memory)
        
        return memories
    
    def extract_from_session(
        self,
        session_history: List[dict],
        min_importance: int = 3
    ) -> List[Memory]:
        """从会话历史中提取记忆"""
        all_memories = []
        
        for message in session_history:
            role = message.get("role", "user")
            content = message.get("content", "")
            
            # 只从用户消息提取
            if role != "user":
                continue
            
            # 提取记忆
            memories = self.extract_from_message(content, role)
            
            # 过滤低重要性
            memories = [m for m in memories if m.importance >= min_importance]
            
            all_memories.extend(memories)
        
        return all_memories
```

#!/usr/bin/env python3
"""记忆搜索 - 支持关键词和语义搜索"""

import logging
from typing import List, Dict, Any
from .memory_manager import Memory, MemoryManager

logger = logging.getLogger(__name__)


class MemorySearch:
    """记忆搜索引擎"""
    
    def __init__(self, memory_manager: MemoryManager):
        self.memory_manager = memory_manager
        self.use_vector_search = False
    
    def search(
        self,
        query: str,
        category: str = None,
        tags: List[str] = None,
        top_k: int = 5,
        use_vector: bool = False
    ) -> List[Dict[str, Any]]:
        """
        搜索记忆
        
        Args:
            query: 搜索词
            category: 分类过滤
            tags: 标签过滤
            top_k: 返回数量
            use_vector: 是否使用向量搜索
        
        Returns:
            搜索结果列表
        """
        if use_vector and self.use_vector_search:
            return self._vector_search(query, top_k)
        else:
            return self._keyword_search(query, category, tags, top_k)
    
    def _keyword_search(
        self,
        query: str,
        category: str,
        tags: List[str],
        top_k: int
    ) -> List[Dict[str, Any]]:
        """关键词搜索"""
        memories = list(self.memory_manager.memories.values())
        
        # 分类过滤
        if category:
            memories = [m for m in memories if m.category == category]
        
        # 标签过滤
        if tags:
            memories = [m for m in memories if any(t in m.tags for t in tags)]
        
        # 关键词匹配（简单文本匹配）
        query_lower = query.lower()
        scored_memories = []
        
        for memory in memories:
            score = 0
            
            # 内容匹配
            if query_lower in memory.content.lower():
                score += 10
            
            # 标签匹配
            if any(query_lower in tag.lower() for tag in memory.tags):
                score += 5
            
            # 重要性加权
            score += memory.importance
            
            # 访问次数加权
            score += min(memory.access_count, 10)
            
            if score > 0:
                scored_memories.append((score, memory))
        
        # 按分数排序
        scored_memories.sort(key=lambda x: x[0], reverse=True)
        
        # 返回 top_k
        results = []
        for score, memory in scored_memories[:top_k]:
            result = memory.to_dict()
            result["score"] = score
            results.append(result)
        
        return results

#!/usr/bin/env python3
"""搜索记忆技能"""

from typing import Dict, Any
from lingxi.core.memory import MemoryManager, MemorySearch


def execute(parameters: Dict[str, Any]) -> str:
    """
    执行记忆搜索
    
    参数：
    - query: 搜索关键词（必填）
    - category: 分类过滤（可选）
    - tags: 标签过滤（可选）
    - top_k: 返回数量（可选，默认 5）
    
    返回：
    - 格式化的搜索结果
    """
    query = parameters.get("query")
    if not query:
        return "错误：缺少 query 参数"
    
    category = parameters.get("category")
    tags = parameters.get("tags", [])
    top_k = parameters.get("top_k", 5)
    
    # 获取记忆管理器
    memory_manager = MemoryManager()
    search_engine = MemorySearch(memory_manager)
    
    # 搜索
    results = search_engine.search(
        query=query,
        category=category,
        tags=tags,
        top_k=top_k
    )
    
    if not results:
        return f"未找到关于 '{query}' 的记忆"
    
    # 格式化结果
    formatted = []
    formatted.append(f"📚 找到 {len(results)} 条相关记忆：\n")
    
    for i, result in enumerate(results, 1):
        formatted.append(f"{i}. 【{result['category']}】(重要性：{'⭐' * result['importance']})")
        formatted.append(f"   {result['content']}")
        if result.get('tags'):
            formatted.append(f"   标签：{', '.join(result['tags'])}")
        formatted.append("")
    
    return "\n".join(formatted)
```

        ]
        
        if not important_memories:
            return ""
        
        lines = []
        for memory in important_memories[:10]:  # 最多 10 条
            lines.append(f"- {memory.content}")
        
        return "\n".join(lines)

            memories = self.memory_search.search(task, top_k=3)
            if memories:
                # 注入记忆到上下文
                context = context or {}
                context["retrieved_memories"] = memories

        return await super().execute_task(task, session_id, context)
    
    def _should_search_memory(self, task: str) -> bool:

        keywords = ["我记得", "你记得", "之前说过", "上次", "以前", "偏好", "习惯"]
        return any(kw in task for kw in keywords)

### 7.2 解析器

```python
# lingxi/core/memory/memory_parser.py

class MemoryParser:
    """MEMORY.md 解析器"""
    
    def parse(self, content: str) -> List[Memory]:
        """解析 MEMORY.md 内容"""
        memories = []
        
        # 按章节分割
        sections = self._split_sections(content)
        
        for section_name, section_content in sections.items():
            category = self._map_category(section_name)
            
            # 提取列表项
            items = self._extract_list_items(section_content)
            
            for item in items:
                memory = Memory(
                    id=Memory.compute_id(item, category),
                    content=item,
                    category=category,
                    tags=[],
                    importance=3,
                    created_at=datetime.now().timestamp(),
                    updated_at=datetime.now().timestamp(),
                    access_count=0,
                    workspace_path="",
                    metadata={"source": "MEMORY.md", "section": section_name}
                )
                memories.append(memory)
        
        return memories
    
    def _map_category(self, section_name: str) -> str:
        """映射章节名到分类"""
        mapping = {
            "用户偏好": "preference",
            "重要事实": "fact",
            "历史决策": "decision",
            "待办事项": "todo"
        }
        return mapping.get(section_name, "note")
```

## 9. 使用示例

### 9.1 手动保存记忆

```python
from lingxi.core.memory import MemoryManager

manager = MemoryManager()

# 保存用户偏好
manager.save_memory(
    content="我喜欢简洁的代码风格",
    category="preference",
    tags=["coding", "style"],
    importance=4
)

# 保存重要事实
manager.save_memory(
    content="当前项目是灵犀助手 v2.0",
    category="fact",
    tags=["project"],
    importance=5
)
```

### 9.2 搜索记忆

```python
from lingxi.core.memory import MemorySearch, MemoryManager

manager = MemoryManager()
search = MemorySearch(manager)

# 关键词搜索
results = search.search("代码风格", top_k=5)

for result in results:
    print(f"{result['category']}: {result['content']}")
```
