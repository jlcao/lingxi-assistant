class TaskClassifier:
    def init(self, config_path: str = "config.yaml"):
        self.config = load_yaml(config_path)
        self.llm_client = LLMClient(config.llm)
        
    def classify(self, task_text: str) -> dict:
        """
        简化分类：优先LLM分类，规则作为fallback
        """
        # Step 1: LLM分类（主要方式）
        try:
            llm_result = self._llm_classify(task_text)
            if llm_result["confidence"] >= 0.7:
                return llm_result
        except Exception as e:
            log_warning(f"LLM分类失败：{e}")
        
        # Step 2: 规则fallback（LLM不可用时）
        return self._rule_classify(task_text)
    
    def _llm_classify(self, task_text: str) -> dict:
        """LLM分类实现"""
        prompt = f"""
你是任务分类器，将用户请求分为三级：
trivial：无需工具，直接回答（如问候、简单问答）
simple：单一步骤、单工具调用（如查天气、翻译）
complex：多步骤、多工具调用（如旅行规划、数据分析）

输出JSON格式：
{{"level": "trivial|simple|complex", "confidence": 0.0-1.0, "reason": "理由"}}

用户请求：{task_text}
"""
        response = self.llm_client.generate(prompt, max_tokens=200)
        return parse_json(response)
    
    def _rule_classify(self, task_text: str) -> dict:
        """规则分类fallback"""
        # 简单关键词匹配
        complex_keywords = ["规划", "分析", "对比", "总结", "多步骤"]
        trivial_keywords = ["你好", "谢谢", "再见", "是谁", "什么是"]
        
        if any(kw in task_text for kw in trivial_keywords):
            return {"level": "trivial", "confidence": 0.8, "reason": "规则匹配"}
        elif any(kw in task_text for kw in complex_keywords):
            return {"level": "complex", "confidence": 0.7, "reason": "规则匹配"}
        else:
            return {"level": "simple", "confidence": 0.6, "reason": "默认"}

个人版会话管理（SQLite + 内存）
class SessionManager:
    def init(self, db_path: str = "data/assistant.db"):
        self.db_path = db_path
        self.memory_cache = {}  # 内存缓存活跃会话
        self._init_db()
        
    def _init_db(self):
        """初始化SQLite数据库"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS sessions (
                session_id TEXT PRIMARY KEY,
                user_name TEXT DEFAULT 'default',
                history_json TEXT,
                checkpoint_json TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        conn.commit()
        conn.close()
    
    def get_history(self, session_id: str, max_turns: int = 20) -> list:
        """获取会话历史（最近N轮）"""
        # 优先从内存缓存获取
        if session_id in self.memory_cache:
            return self.memory_cache[session_id][-max_turns:]
        
        # 从数据库加载
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("SELECT history_json FROM sessions WHERE session_id = ?", 
                      (session_id,))
        row = cursor.fetchone()
        conn.close()
        
        if row and row[0]:
            history = json.loads(row[0])
            self.memory_cache[session_id] = history
            return history[-max_turns:]
        return []
    
    def add_turn(self, session_id: str, role: str, content: str):
        """添加对话轮次"""
        history = self.get_history(session_id)
        history.append({"role": role, "content": content, "time": time.time()})
        
        # 保留最近50轮
        if len(history) > 50:
            history = history[-50:]
        
        # 更新内存缓存
        self.memory_cache[session_id] = history
        
        # 异步写入数据库
        self._save_to_db(session_id, history)
    
    def _save_to_db(self, session_id: str, history: list):
        """保存到SQLite"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            INSERT OR REPLACE INTO sessions (session_id, history_json, updated_at)
            VALUES (?, ?, CURRENT_TIMESTAMP)
        """, (session_id, json.dumps(history, ensure_ascii=False)))
        conn.commit()
        conn.close()
    
    def save_checkpoint(self, session_id: str, state: dict):
        """保存执行检查点（用于断点续传）"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            INSERT OR REPLACE INTO sessions (session_id, checkpoint_json, updated_at)
            VALUES (?, ?, CURRENT_TIMESTAMP)
        """, (session_id, json.dumps(state, ensure_ascii=False)))
        conn.commit()
        conn.close()
    
    def restore_checkpoint(self, session_id: str) -> dict:
        """恢复检查点"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("SELECT checkpoint_json FROM sessions WHERE session_id = ?", 
                      (session_id,))
        row = cursor.fetchone()
        conn.close()
        
        if row and row[0]:
            return json.loads(row[0])
        return None
    
    def clear_session(self, session_id: str):
        """清空会话"""
        if session_id in self.memory_cache:
            del self.memory_cache[session_id]
        
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("DELETE FROM sessions WHERE session_id = ?", (session_id,))
        conn.commit()
        conn.close()
    
    def get_checkpoint_status(self, session_id: str) -> dict:
        """获取检查点状态信息"""
        checkpoint = self.restore_checkpoint(session_id)
        if not checkpoint:
            return {"has_checkpoint": False}
        
        return {
            "has_checkpoint": True,
            "task": checkpoint.get("task"),
            "current_step": checkpoint.get("current_step_idx", 0),
            "total_steps": len(checkpoint.get("plan", [])),
            "execution_status": checkpoint.get("execution_status"),
            "replan_count": checkpoint.get("replan_count", 0),
            "timestamp": checkpoint.get("timestamp"),
            "error_info": checkpoint.get("error_info")
        }
    
    def clear_checkpoint(self, session_id: str):
        """清除指定会话的检查点"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            UPDATE sessions SET checkpoint_json = NULL, updated_at = CURRENT_TIMESTAMP
            WHERE session_id = ?
        """, (session_id,))
        conn.commit()
        conn.close()
    
    def cleanup_expired_checkpoints(self, ttl_hours: int = 24):
        """清理过期的检查点"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 获取所有会话
        cursor.execute("SELECT session_id, checkpoint_json, updated_at FROM sessions")
        sessions = cursor.fetchall()
        
        current_time = time.time()
        ttl_seconds = ttl_hours * 3600
        cleaned_count = 0
        
        for session_id, checkpoint_json, updated_at in sessions:
            if checkpoint_json:
                try:
                    checkpoint = json.loads(checkpoint_json)
                    checkpoint_time = checkpoint.get("timestamp", updated_at)
                    
                    # 检查是否过期
                    if current_time - checkpoint_time > ttl_seconds:
                        # 清除过期检查点
                        cursor.execute("""
                            UPDATE sessions SET checkpoint_json = NULL, updated_at = CURRENT_TIMESTAMP
                            WHERE session_id = ?
                        """, (session_id,))
                        cleaned_count += 1
                        log_info(f"清除过期检查点：{session_id}")
                except Exception as e:
                    log_error(f"处理检查点{session_id}时出错：{e}")
        
        conn.commit()
        conn.close()
        
        if cleaned_count > 0:
            log_info(f"清理了{cleaned_count}个过期检查点")
        
        return cleaned_count
    
    def list_active_checkpoints(self) -> list:
        """列出所有活跃的检查点"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            SELECT session_id, checkpoint_json, updated_at 
            FROM sessions 
            WHERE checkpoint_json IS NOT NULL
        """)
        rows = cursor.fetchall()
        conn.close()
        
        checkpoints = []
        for session_id, checkpoint_json, updated_at in rows:
            try:
                checkpoint = json.loads(checkpoint_json)
                checkpoints.append({
                    "session_id": session_id,
                    "task": checkpoint.get("task"),
                    "current_step": checkpoint.get("current_step_idx", 0),
                    "total_steps": len(checkpoint.get("plan", [])),
                    "execution_status": checkpoint.get("execution_status"),
                    "updated_at": updated_at
                })
            except Exception as e:
                log_error(f"解析检查点{session_id}时出错：{e}")
        
        return checkpoints

class ExecutionModeSelector:
    def __init__(self, config: dict):
        self.config = config
        self.llm_config = config.get("llm", {})
        self.models_config = self.llm_config.get("models", {})
        self.default_model = self.llm_config.get("default_model", "qwen-plus")

    def select_model(self, task_level: str) -> str:
        """根据任务级别选择模型

        Args:
            task_level: 任务级别（trivial/simple/complex）

        Returns:
            模型名称
        """
        if task_level in self.models_config:
            model_config = self.models_config[task_level]
            return model_config.get("model", self.default_model)
        return self.default_model

    def get_model_config(self, task_level: str) -> dict:
        """获取任务级别对应的模型配置

        Args:
            task_level: 任务级别

        Returns:
            模型配置字典
        """
        if task_level in self.models_config:
            return self.models_config[task_level]
        return {"model": self.default_model}

3.4.2 Plan+ReAct 模式执行流程

class PlanReActEngine:
    """支持断点重试的Plan+ReAct执行引擎"""
    
    def init(self, config: dict, skill_registry: SkillRegistry, 
                 session_manager: SessionManager):
        self.config = config
        self.skill_registry = skill_registry
        self.session_manager = session_manager
        self.llm_client = LLMClient(config.llm)
        self.max_plan_steps = config.execution_mode.complex.max_plan_steps
        self.max_replan_count = config.execution_mode.complex.max_replan_count
        self.max_step_retries = config.execution_mode.complex.get("max_step_retries", 3)
        
    def execute(self, task: str, session_id: str, history: list = None) -> dict:
        """执行Plan+ReAct流程，支持断点恢复"""
        
        # Step 1: 检查是否存在检查点
        existing_checkpoint = self.session_manager.restore_checkpoint(session_id)
        
        if existing_checkpoint and existing_checkpoint.get("task") == task:
            # 从检查点恢复执行
            return self._resume_from_checkpoint(existing_checkpoint, history)
        else:
            # 新任务执行
            return self._execute_new_task(task, session_id, history)
    
    def _execute_new_task(self, task: str, session_id: str, history: list = None) -> dict:
        """执行新任务"""
        log_info(f"开始执行新任务：{task}")
        
        # Step 1: 生成Plan
        plan = self._generate_plan(task, history)
        if not plan:
            return {"success": False, "error": "Plan生成失败"}
        
        log_info(f"生成Plan，共{len(plan)}个步骤")
        
        # Step 2: 保存初始检查点
        initial_checkpoint = {
            "task": task,
            "plan": plan,
            "current_step_idx": 0,
            "completed_steps": [],
            "step_results": [],
            "replan_count": 0,
            "execution_status": "running",
            "error_info": None,
            "timestamp": time.time()
        }
        self.session_manager.save_checkpoint(session_id, initial_checkpoint)
        
        # Step 3: 执行每个步骤
        return self._execute_steps(plan, session_id, history, initial_checkpoint)
    
    def _resume_from_checkpoint(self, checkpoint: dict, history: list = None) -> dict:
        """从检查点恢复执行"""
        log_info(f"从检查点恢复执行，当前步骤：{checkpoint['current_step_idx']}")
        
        plan = checkpoint["plan"]
        current_step_idx = checkpoint["current_step_idx"]
        completed_steps = checkpoint["completed_steps"]
        step_results = checkpoint["step_results"]
        replan_count = checkpoint.get("replan_count", 0)
        
        # 从当前步骤继续执行
        updated_checkpoint = checkpoint.copy()
        updated_checkpoint["execution_status"] = "running"
        self.session_manager.save_checkpoint(session_id, updated_checkpoint)
        
        return self._execute_steps(
            plan, 
            session_id, 
            history, 
            updated_checkpoint,
            start_idx=current_step_idx,
            completed_steps=completed_steps,
            step_results=step_results,
            replan_count=replan_count
        )
    
    def _execute_steps(self, plan: list, session_id: str, history: list, 
                      checkpoint: dict, start_idx: int = 0, 
                      completed_steps: list = None, step_results: list = None,
                      replan_count: int = 0) -> dict:
        """执行步骤序列"""
        
        if completed_steps is None:
            completed_steps = []
        if step_results is None:
            step_results = []
        
        all_results = step_results
        
        # 从指定步骤开始执行
        for step_idx in range(start_idx, len(plan)):
            step = plan[step_idx]
            
            # 检查是否需要重规划
            if replan_count >= self.max_replan_count:
                log_warning("达到最大重规划次数，继续执行原Plan")
            
            # 执行当前步骤（支持重试）
            step_result = self._execute_step_with_retry(
                step, step_idx, session_id, history
            )
            
            if step_result["success"]:
                # 步骤成功，更新检查点
                completed_steps.append({
                    "step": step_idx + 1,
                    "description": step["description"]
                })
                
                all_results.append({
                    "step": step_idx + 1,
                    "description": step["description"],
                    "result": step_result["result"]
                })
                
                # 更新检查点
                self.session_manager.save_checkpoint(session_id, {
                    "task": checkpoint["task"],
                    "plan": plan,
                    "current_step_idx": step_idx + 1,
                    "completed_steps": completed_steps,
                    "step_results": all_results,
                    "replan_count": replan_count,
                    "execution_status": "running",
                    "error_info": None,
                    "timestamp": time.time()
                })
                
                log_info(f"步骤{step_idx + 1}执行成功")
            else:
                # 步骤失败，尝试重规划
                error_msg = step_result.get('error', '未知错误')
                log_warning(f"步骤{step_idx + 1}失败：{error_msg}")
                
                if replan_count < self.max_replan_count:
                    log_info(f"尝试重规划（{replan_count + 1}/{self.max_replan_count}）")
                    
                    # 生成修正后的Plan
                    new_plan = self._replan(
                        checkpoint["task"],
                        plan,
                        step_idx,
                        step_result,
                        history
                    )
                    
                    if new_plan:
                        log_info("重规划成功，使用新Plan继续执行")
                        replan_count += 1
                        
                        # 更新检查点，使用新Plan
                        self.session_manager.save_checkpoint(session_id, {
                            "task": checkpoint["task"],
                            "plan": new_plan,
                            "current_step_idx": step_idx,
                            "completed_steps": completed_steps,
                            "step_results": all_results,
                            "replan_count": replan_count,
                            "execution_status": "replanned",
                            "error_info": {"step": step_idx + 1, "error": error_msg},
                            "timestamp": time.time()
                        })
                        
                        # 使用新Plan继续执行
                        return self._execute_steps(
                            new_plan, 
                            session_id, 
                            history, 
                            checkpoint,
                            start_idx=step_idx,
                            completed_steps=completed_steps,
                            step_results=all_results,
                            replan_count=replan_count
                        )
                    else:
                        log_error("重规划失败，终止执行")
                        return {
                            "success": False,
                            "error": f"步骤{step_idx + 1}失败且重规划失败",
                            "partial_results": all_results
                        }
                else:
                    log_error("达到最大重规划次数，终止执行")
                    return {
                        "success": False,
                        "error": f"步骤{step_idx + 1}失败，达到最大重规划次数",
                        "partial_results": all_results
                    }
        
        # 所有步骤执行完成
        log_info("所有步骤执行完成，开始整合结果")
        final_result = self._synthesize_results(
            checkpoint["task"],
            all_results,
            history
        )
        
        # 清除检查点
        self.session_manager.save_checkpoint(session_id, {
            "task": checkpoint["task"],
            "plan": plan,
            "current_step_idx": len(plan),
            "completed_steps": completed_steps,
            "step_results": all_results,
            "replan_count": replan_count,
            "execution_status": "completed",
            "error_info": None,
            "timestamp": time.time()
        })
        
        return {
            "success": True,
            "result": final_result,
            "steps_executed": len(all_results),
            "replan_count": replan_count
        }
    
    def _execute_step_with_retry(self, step: dict, step_idx: int, 
                                  session_id: str, history: list) -> dict:
        """执行单个步骤，支持重试"""
        for attempt in range(self.max_step_retries):
            try:
                log_info(f"执行步骤{step_idx + 1}（尝试{attempt + 1}/{self.max_step_retries}）")
                
                react_engine = ReActEngine(self.config, self.skill_registry)
                step_result = react_engine.execute(
                    task=step["description"],
                    session_id=session_id,
                    history=history
                )
                
                if step_result["success"]:
                    return step_result
                else:
                    log_warning(f"步骤{step_idx + 1}尝试{attempt + 1}失败：{step_result.get('error')}")
                    
            except Exception as e:
                log_error(f"步骤{step_idx + 1}尝试{attempt + 1}异常：{str(e)}")
        
        # 所有重试都失败
        return {
            "success": False,
            "error": f"步骤{step_idx + 1}执行失败，达到最大重试次数{self.max_step_retries}"
        }
    
    def _generate_plan(self, task: str, history: list = None) -> list:
        prompt = f"""
请为以下任务生成执行步骤清单（JSON数组格式）：
任务：{task}

输出格式：
[
  {{"step": 1, "description": "步骤1描述", "expected_tool": "工具ID或null"}},
  {{"step": 2, "description": "步骤2描述", "expected_tool": "工具ID或null"}}
]

步骤数量不超过{self.max_plan_steps}步。
"""
        response = self.llm_client.generate(prompt, max_tokens=1000)
        return parse_json_list(response)
    
    def _replan(self, task: str, original_plan: list, failed_step_idx: int, 
                failure_info: dict, history: list = None) -> list:
        """生成修正后的Plan"""
        prompt = f"""
原任务：{task}
原Plan执行到步骤{failed_step_idx + 1}时失败。
失败信息：{failure_info.get('error', '未知错误')}
已完成的步骤：{original_plan[:failed_step_idx]}

请生成修正后的执行Plan（JSON数组格式），考虑失败原因调整后续步骤。
"""
        response = self.llm_client.generate(prompt, max_tokens=1000)
        return parse_json_list(response)
    
    def _synthesize_results(self, task: str, step_results: list, 
                           history: list = None) -> str:
        """整合所有步骤结果为最终答案"""
        prompt = f"""
任务：{task}
已执行的步骤结果：
{json.dumps(step_results, ensure_ascii=False, indent=2)}

请整合以上结果，生成用户友好的最终答案。
"""
        response = self.llm_client.generate(prompt, max_tokens=2000)
        return response

class SkillRegistry:
    """本地技能注册表（SQLite）"""
    
    def init(self, db_path: str = "data/skills.db"):
        self.db_path = db_path
        self._init_db()
        self._load_to_cache()
    
    def _init_db(self):
        """初始化数据库"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS skills (
                skill_id TEXT PRIMARY KEY,
                skill_name TEXT,
                version TEXT DEFAULT 'v1',
                api_url TEXT,
                api_key TEXT,
                input_schema TEXT,
                description TEXT,
                enabled INTEGER DEFAULT 1,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        conn.commit()
        conn.close()
    
    def _load_to_cache(self):
        """加载技能到内存缓存"""
        self.skill_cache = {}
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("SELECT * FROM skills WHERE enabled = 1")
        for row in cursor.fetchall():
            skill = {
                "skill_id": row[0],
                "skill_name": row[1],
                "version": row[2],
                "api_url": row[3],
                "api_key": row[4],
                "input_schema": json.loads(row[5]) if row[5] else None,
                "description": row[6]
            }
            self.skill_cache[skill["skill_id"]] = skill
        conn.close()
    
    def register(self, skill: dict) -> bool:
        """注册新技能"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        cursor.execute("""
            INSERT OR REPLACE INTO skills 
            (skill_id, skill_name, version, api_url, api_key, input_schema, description)
            VALUES (?, ?, ?, ?, ?, ?, ?)
        """, (
            skill["skill_id"],
            skill["skill_name"],
            skill.get("version", "v1"),
            skill["api_url"],
            skill.get("api_key", ""),
            json.dumps(skill.get("input_schema")),
            skill.get("description", "")
        ))
        conn.commit()
        conn.close()
        self._load_to_cache()  # 刷新缓存
        return True
    
    def get_skill(self, skill_id: str) -> dict:
        """获取技能信息"""
        return self.skill_cache.get(skill_id)
    
    def call(self, skill_id: str, parameters: dict) -> dict:
        """调用技能"""
        skill = self.get_skill(skill_id)
        if not skill:
            return {"success": False, "error": f"技能{skill_id}不存在"}
        
        if not skill["enabled"]:
            return {"success": False, "error": f"技能{skill_id}已禁用"}
        
        # 简化调用逻辑
        try:
            headers = {}
            if skill["api_key"]:
                headers["Authorization"] = f"Bearer {skill['api_key']}"
            
            response = requests.post(
                skill["api_url"],
                json=parameters,
                headers=headers,
                timeout=30
            )
            
            if response.status_code == 200:
                return {"success": True, "data": response.json()}
            else:
                return {"success": False, "error": f"HTTP {response.status_code}: {response.text}"}
                
        except Exception as e:
            return {"success": False, "error": str(e)}
    
    def list_skills(self) -> list:
        """列出所有可用技能"""
        return [
            {"skill_id": s["skill_id"], "skill_name": s["skill_name"], 
             "description": s["description"]}
            for s in self.skill_cache.values()
        ]

def process_task(task: str, session_id: str, config: dict, 
                 classifier, skill_registry, session_manager) -> str:
    """处理单个任务"""
    # 1. 任务分类
    classification = classifier.classify(task)
    level = classification["level"]
    log_info(f"任务分类：{level} (置信度：{classification['confidence']})")
    
    # 2. 获取会话历史
    history = session_manager.get_history(session_id)
    
    # 3. 选择执行引擎
    if level == "trivial":
        # 直接回答
        llm = LLMClient(config.llm)
        result = llm.generate(task, max_tokens=1000)
    elif level == "simple":
        # ReAct模式
        engine = ReActEngine(config, skill_registry)
        result_dict = engine.execute(task, session_id, history)
        result = result_dict.get("result", result_dict.get("error", "执行失败"))
    else:  # complex
        # Plan+ReAct模式
        engine = PlanReActEngine(config, skill_registry, session_manager)
        result_dict = engine.execute(task, session_id, history)
        result = result_dict.get("result", result_dict.get("error", "执行失败"))
    
    # 4. 保存会话
    session_manager.add_turn(session_id, "user", task)
    session_manager.add_turn(session_id, "assistant", result)
    
    # 5. 记录执行日志
    log_execution(task, level, result, session_id)
    
    return result

七、断点重试功能说明

7.1 功能概述

Plan+ReAct模式支持完整的步骤断点重试功能，确保在系统崩溃、网络中断或其他异常情况下，可以从上次失败的位置继续执行，而不是重新开始整个任务。

7.2 检查点机制

**检查点数据结构：**
```python
checkpoint = {
    "task": "原始任务描述",
    "plan": [步骤1, 步骤2, ...],           # 执行计划
    "current_step_idx": 2,                 # 当前步骤索引
    "completed_steps": [步骤1, 步骤2],      # 已完成的步骤
    "step_results": [结果1, 结果2],        # 步骤执行结果
    "replan_count": 1,                     # 重规划次数
    "execution_status": "running",         # 执行状态
    "error_info": None,                    # 错误信息
    "timestamp": 1234567890.0             # 时间戳
}
```

**执行状态：**
- `running`: 正在执行
- `replanned`: 已重规划
- `completed`: 执行完成
- `failed`: 执行失败

7.3 自动断点恢复

系统在执行Plan+ReAct任务时会自动：
1. 在开始执行时保存初始检查点
2. 每完成一个步骤后更新检查点
3. 在重规划时保存新的Plan和状态
4. 在任务完成时标记为completed状态

当任务中断后重新执行相同任务时，系统会：
1. 检查是否存在该任务的检查点
2. 如果存在且任务匹配，从上次失败的位置继续执行
3. 如果不存在或任务不匹配，开始新任务执行

7.6 使用场景

**场景1：网络中断恢复**
```

系统：检测到检查点，从步骤2继续执行...
```

**场景2：步骤失败重试**
```
用户：查询天气并翻译成英文
系统：步骤1/2：查询天气...成功
系统：步骤2/2：翻译...失败
系统：重试步骤2（1/3）...失败
系统：重试步骤2（2/3）...失败
系统：重试步骤2（3/3）...成功
系统：任务完成
```

**场景3：重规划恢复**
```

系统：步骤1/5：查询航班...失败

```

7.7 注意事项

1. **检查点过期**：默认24小时后自动清理，可通过配置调整
2. **任务匹配**：只有任务描述完全相同时才会恢复检查点
3. **状态一致性**：重规划会生成新的Plan，已完成的步骤结果会保留
4. **资源清理**：定期清理过期检查点，避免占用过多存储空间
5. **并发控制**：同一会话同一时间只能有一个活跃的检查点

from fastapi import FastAPI, WebSocket, WebSocketDisconnect
from fastapi.middleware.cors import CORSMiddleware
import uvicorn

app = FastAPI(title="Lingxi Assistant API", version="3.0.0")

# 允许跨域
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 全局WebSocket连接管理
class ConnectionManager:
    def __init__(self):
        self.active_connections: dict[str, WebSocket] = {}

    async def connect(self, session_id: str, websocket: WebSocket):
        await websocket.accept()
        self.active_connections[session_id] = websocket

    def disconnect(self, session_id: str):
        if session_id in self.active_connections:
            del self.active_connections[session_id]

    async def send_event(self, session_id: str, event: dict):
        if session_id in self.active_connections:
            await self.active_connections[session_id].send_json(event)

    async def broadcast(self, event: dict):
        for connection in self.active_connections.values():
            await connection.send_json(event)

manager = ConnectionManager()

# WebSocket端点
@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket, session_id: str = None):
    await manager.connect(session_id, websocket)
    try:
        while True:
            data = await websocket.receive_text()
            # 处理客户端消息（如心跳、订阅等）
            pass
    except WebSocketDisconnect:
        manager.disconnect(session_id)

# 会话管理API
